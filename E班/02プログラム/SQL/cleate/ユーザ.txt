(
	ユーザID CHAR(7) PRIMARY KEY,
	名前 VARCHAR2(50) NOT NULL,
	フリガナ VARCHAR2(50) NOT NULL,
	性別ID CHAR(2) REFERENCES 性別(性別ID) NOT NULL,
	生年月日 DATE NOT NULL,
	メンバーID CHAR(10) UNIQUE NOT NULL,
	パスワード VARCHAR2(60) NOT NULL,
	アドレス VARCHAR2(60) NOT NULL,
	郵便番号 CHAR(7) NOT NULL,
	出身地 VARCHAR2(200) NOT NULL,
	部署ID CHAR(4) REFERENCES 部署(部署ID) NOT NULL,
	ユーザアイコン VARCHAR2(4000) NOT NULL,
	自己紹介 VARCHAR2(2000),
	権限種別ID CHAR(2) REFERENCES 権限(権限種別ID) NOT NULL
)
