DROP TABLE ユーザ表;


CREATE TABLE ユーザ表
(
 ユーザID CHAR(7) PRIMARY KEY,
 パスワード VARCHAR2(16) NOT NULL,
 姓 VARCHAR2(50) NOT NULL,
 名 VARCHAR2(50) NOT NULL,
 ふりがな_姓 VARCHAR2(50) NOT NULL,
 ふりがな_名 VARCHAR2(50) NOT NULL,
 性別 CHAR(4) NOT NULL,
 生年月日 DATE NOT NULL,
 郵便番号 CHAR(8) NOT NULL,
 メールアドレス VARCHAR2(100) NOT NULL,
 電話番号 VARCHAR2(20) NOT NULL,
 都道府県 VARCHAR2(10) NOT NULL,
 住所 VARCHAR2(120) NOT NULL,
 マンション VARCHAR2(150)
)
